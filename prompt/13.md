type Center =
  | "Head"
  | "Ajna"
  | "Throat"
  | "G"
  | "Heart"
  | "Spleen"
  | "SolarPlexus"
  | "Sacral"
  | "Root"

type Graph = Record<Center, Center[]>

function getConnectedComponents(
  centers: Center[],
  graph: Graph
) {
  const visited = new Set<Center>()
  const components: Center[][] = []

  const dfs = (node: Center, comp: Center[]) => {
    visited.add(node)
    comp.push(node)

    for (const next of graph[node] || []) {
      if (!visited.has(next)) {
        dfs(next, comp)
      }
    }
  }

  for (const c of centers) {
    if (!visited.has(c)) {
      const comp: Center[] = []
      dfs(c, comp)
      components.push(comp)
    }
  }

  return components
}

export function getDefinitionType(
  centers: Center[],
  graph: Graph
) {
  const components =
    getConnectedComponents(centers, graph)

  if (components.length === 1)
    return "Single Definition"

  if (components.length === 2)
    return "Split Definition"

  if (components.length === 3)
    return "Triple Split"

  if (components.length === 4)
    return "Quad Split"

  return "No Definition"
}
